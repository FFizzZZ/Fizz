class Solution:
    def GetNumberOfK(self, data, k):
        if not data: return 0
        first = self.first(data, k)
        last = self.last(data, k)
        return last - first + 1 if first != -1 else 0
    def first(self, A, k):
        start, end = 0, len(A)-1
        while start < end:
            mid = (start+end) // 2
            if A[mid] >= k:
                end = mid
            else:
                start = mid + 1
        return end if A[end] == k else -1
    def last(self, A, k):
        start, end = 0, len(A)-1
        while start < end:
            mid = (start+end) // 2 + 1
            if A[mid] > k:
                end = mid - 1
            else:
                start = mid
        return end if A[end] == k else -1
