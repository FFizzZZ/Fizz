class Solution:
    def PrintFromTopToBottom(self, root):
        # write code here
        ans = []
        def f(node, level):
            if not node: return
            while len(ans) < level:
                ans.append([])
            ans[level-1].append(node.val)
            f(node.left, level+1)
            f(node.right, level+1)
        f(root, 1)
        res = []
        for i in ans:
            res += i
        return res
