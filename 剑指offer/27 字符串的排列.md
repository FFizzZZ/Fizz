class Solution:
    def Permutation(self, ss):
        if not ss: return []
        ans = []
        def f(s, path):
            if not s:
                if path not in ans:
                    ans.append(path)
                return
            for i in range(len(s)):
                f(s[:i]+s[i+1:], path+s[i])
        f(ss, '')
        return ans

class Solution:
    def Permutation(self, ss):
        if not ss: return []
        if len(ss) == 1: return ss
        ans = []
        for i in range(len(ss)):
            for j in self.Permutation(ss[:i]+ss[i+1:]):
                value = ss[i]+ j
                if value not in ans:
                    ans.append(value)
        return ans
