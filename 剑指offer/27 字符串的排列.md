```
class Solution:
    def Permutation(self, ss):
        if not ss: return []
        ans = []
        def f(s, path):
            if not s:
                if path not in ans:
                    ans.append(path)
                return
            for i in range(len(s)):
                f(s[:i]+s[i+1:], path+s[i])
        f(ss, '')
        return ans
```
