class Solution:
    def InversePairs(self, data):
        count = 0
        for item in sorted(data):
            count += data.index(item)
            data.remove(item)             
        return count%1000000007

## Divide and Conquer
```
class Solution:
    def InversePairs(self, data):
        count = self.f(data)[0]
        return count % 1000000007
    def f(self, data):
        if len(data) <= 1: return 0, data
        mid = len(data) / 2
        count1, left = self.f(data[:mid])
        count2, right = self.f(data[mid:])
        count, ans = self.merge(left, right)
        return count1 + count2 + count, ans
    def merge(self, A, B):
        ans, count = [], 0
        j = len(B) - 1
        for i in range(len(A)-1, -1, -1):
            while j > -1:
                if A[i] > B[j]:
                    ans.append(A[i])
                    count += j + 1
                    break
                else:
                    ans.append(B[j])
                    j -= 1
            if j == -1: 
                ans.extend(reversed(A[:i+1]))
                break
        if j > -1:
            ans.extend(reversed(B[:j+1]))
        return count, ans[::-1]
```
