```
class Solution:
    def match(self, s, pattern):
        if not pattern: return not s
        firstmatch = s and pattern[0] in ['.', s[0]]
        if len(pattern) >= 2 and pattern[1] == '*':
            return self.match(s, pattern[2:]) or firstmatch and self.match(s[1:], pattern)
        else:
            return firstmatch and self.match(s[1:], pattern[1:])
```
