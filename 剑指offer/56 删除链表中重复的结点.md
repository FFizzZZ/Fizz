class Solution:
    def deleteDuplication(self, pHead):
        node = ListNode(0)
        node.next = pHead
        p, previous = pHead, node
        while p:
            if p.next and p.next.val == p.val:
                while p.next and p.next.val == p.val:
                    p = p.next
                p = p.next
                previous.next = p
            else:
                previous = p
                p = p.next
        return node.next
