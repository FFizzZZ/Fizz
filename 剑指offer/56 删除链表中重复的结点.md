class Solution:
    def deleteDuplication(self, pHead):
        pre = head = ListNode(0)
        node = pre.next = pHead
        while node:
            if node.next and node.next.val == node.val:
                while node.next and node.next.val == node.val:
                    node = node.next
                node = node.next
                pre.next = node
            else:
                pre = node
                node = node.next
        return head.next

class Solution:

        node = ListNode(0)

        p, previous = pHead, node
        while p:
            if p.next and p.next.val == p.val:
                while p.next and p.next.val == p.val:
                    p = p.next
                p = p.next
                previous.next = p
            else:
                previous = p
                p = p.next
        return node.next

## Recursion
```
class Solution:
    def deleteDuplication(self, pHead):
        if not pHead: return None
        if pHead.next and pHead.next.val == pHead.val:
            while pHead.next and pHead.next.val == pHead.val:
                pHead = pHead.next
            pHead = pHead.next
            return self.deleteDuplication(pHead)
        else:
            pHead.next = self.deleteDuplication(pHead.next)
            return pHead
```
