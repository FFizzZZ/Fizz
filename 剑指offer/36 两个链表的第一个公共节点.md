## 不需要辅助栈
```
class Solution:
    def FindFirstCommonNode(self, pHead1, pHead2):
        p1, p2, count1, count2 = pHead1, pHead2, 0, 0
        while p1:
            p1 = p1.next
            count1 += 1
        while p2:
            p2 = p2.next
            count2 += 1
        if count1 > count2: pHead1, pHead2 = pHead2, pHead1
        for i in range(abs(count2-count1)): pHead2 = pHead2.next
        while pHead1:
            if pHead1 == pHead2:
                return pHead1
            else:
                pHead1, pHead2 = pHead1.next, pHead2.next
        return None
```

## 辅助栈
```
class Solution:
    def FindFirstCommonNode(self, pHead1, pHead2):
        m, n = [], []
        while pHead1:
            m.append(pHead1)
            pHead1 = pHead1.next
        while pHead2:
            n.append(pHead2)
            pHead2 = pHead2.next
        l = min(len(m), len(n))
        i = 1
        for i in range(1, l+1):
            if m[-i] != n[-i]: break
            if i == l: return n[0]
        return n[-i+1] if i > 1 else None
```
