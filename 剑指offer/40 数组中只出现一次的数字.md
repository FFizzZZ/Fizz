class Solution:
    # 返回[a,b] 其中ab是出现一次的两个数字
    def FindNumsAppearOnce(self, array):
        # write code here
        temp = array[0]
        for i in array[1:]:
            temp ^= i
        temp = bin(temp)
        for i in range(-1, -len(temp)-1, -1):
            if temp[i] == '1':
                break
        left, right = [], []
        for j in array:
            if bin(j)[i] == '1':
                left.append(j)
            else:
                right.append(j)
        print(left)
        print(right)
        l, r = left[0], right[0]
        for i in left[1:]:
            l ^= i
        for i in right[1:]:
            r ^= i
        return [l, r]
