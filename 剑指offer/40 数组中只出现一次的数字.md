```
class Solution:
    def FindNumsAppearOnce(self, array):
        cur = 0
        for x in array:
            cur ^= x
        idx = 0
        while True:
            if cur & 1:
                break
            cur >>= 1
            idx += 1
        left = right = 0
        for x in array:
            digit = x >> idx
            if digit & 1:
                left ^= x
            else:
                right ^= x
        return left, right
```
