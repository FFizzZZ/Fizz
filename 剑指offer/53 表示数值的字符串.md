class Solution:
    def isNumeric(self, s):
        if not s: return False
        s = s[1:] if s[0] in ['+', '-'] else s
        d = ['e', 'E', '.'] + list('0123456789')
        dot = 1
        for i, v in enumerate(s):
            if v not in d: return False
            else:
                if ord('0') <= ord(v) <= ord('9'): continue
                if v == 'e' or v == 'E':
                    return self.isNumeric(s[i+1:]) and '.' not in s[i+1:]
                if v == '.':
                    dot -= 1
                    if dot < 0: return False
        return True
