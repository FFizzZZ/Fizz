class Solution:
    # array 二维列表
    def Find(self, target, array):
        # write code here
        def binary(num, target):
            left, right = 0, len(num) - 1
            while left <= right:
                mid = (left + right) / 2
                if num[mid] == target:
                    return True
                elif num[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            return False
        for i in array:
            if binary(i, target):
                return True
        return False
