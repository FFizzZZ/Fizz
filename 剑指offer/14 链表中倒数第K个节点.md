class Solution:
    def FindKthToTail(self, head, k):
        # write code here
        if not head: return None
        first, second = head, head
        for i in range(k):
            second = second.next
            if not second and i < k - 1: return None
        while second:
            second = second.next
            first = first.next
        return first
