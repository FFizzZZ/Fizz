class Solution:
    def KthNode(self, pRoot, k):
        # write code here
        if not pRoot: return None
        stack, count = [], 0
        while stack or pRoot:
            while pRoot:
                stack.append(pRoot)
                pRoot = pRoot.left
            pRoot = stack.pop()
            count += 1
            if count == k: return pRoot
            pRoot = pRoot.right
