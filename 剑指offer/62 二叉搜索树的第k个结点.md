class Solution:
    def KthNode(self, pRoot, k):
        count = 0
        stack = []
        while pRoot or stack:
            while pRoot:
                stack.append(pRoot)
                pRoot = pRoot.left
            pRoot = stack.pop()
            count += 1
            if count == k: return pRoot
            pRoot = pRoot.right

class Solution:
    def KthNode(self, pRoot, k):
        # write code here
        if not pRoot or not k: return None
        self.count, self.flag = 0, 0
        self.Inorder(pRoot, k)
        return self.count if self.flag == 1 else None
    def Inorder(self, node, k):
        if not node or self.flag: return
        self.Inorder(node.left, k)
        if self.flag: return
        self.count += 1
        if self.count == k:
            self.count = node
            self.flag = 1
        self.Inorder(node.right, k)

class Solution:
    # 返回对应节点TreeNode
    def KthNode(self, pRoot, k):
        self.res=[]
        self.dfs(pRoot)
        return self.res[k-1] if 0<k<=len(self.res) else None
    def dfs(self,root):
        if not root:return
        self.dfs(root.left)
        self.res.append(root)
        self.dfs(root.right)
