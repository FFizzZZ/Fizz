class Solution:
    def Print(self, pRoot):
        if not pRoot: return []
        ans = []
        stack = [pRoot]
        while stack:
            val, temp = [], []
            for i in stack:
                val.append(i.val)
                if i.left: temp.append(i.left)
                if i.right: temp.append(i.right)
            ans.append(val)
            stack = temp
        return ans

class Solution:
    def Print(self, pRoot):
        if not pRoot: return []
        ans, val = [], []
        stack = [pRoot]
        count = 1
        while stack:
            temp = 0
            for i in range(count):
                node = stack.pop(0)
                val.append(node.val)
                if node.left: 
                    stack.append(node.left)
                    temp += 1
                if node.right: 
                    stack.append(node.right)
                    temp += 1
            ans.append(val)
            val = []
            count = temp
        return ans
