class Solution:
    def Print(self, pRoot):
        if not pRoot: return []
        ans = []
        stack = [pRoot]
        while stack:
            val, temp = [], []
            for i in stack:
                val.append(i.val)
                if i.left: temp.append(i.left)
                if i.right: temp.append(i.right)
            ans.append(val)
            stack = temp
        return ans
