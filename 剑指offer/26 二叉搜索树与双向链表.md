## stack
```
class Solution:
    def Convert(self, root):
        if not root: return None
        stack = []
        pre = head = TreeNode(0)
        while root or stack:
            while root:
                stack.append(root)
                root = root.left
            root = stack.pop()
            root.left = pre
            pre.right = root
            pre = root
            root = root.right
        head.right.left = None
        return head.right
```

## recursive
```
class Solution:
    def Convert(self, root):
        if not root: return None
        self.head = self.pre = TreeNode(0)
        def dfs(node):
            if not node: return
            dfs(node.left)
            self.pre.right = node
            node.left = self.pre
            self.pre = node
            dfs(node.right)
        dfs(root)
        ans = self.head.right
        ans.left = None
        return ans
```

class Solution:
    def Convert(self, pRootOfTree):
        if not pRootOfTree:
            return
        # 处理左子树
        self.Convert(pRootOfTree.left)
        left=pRootOfTree.left
        # 连接根与左子树最大结点
        if left:
            while(left.right):
                left=left.right
            pRootOfTree.left,left.right=left,pRootOfTree
        # 处理右子树
        self.Convert(pRootOfTree.right)
        right=pRootOfTree.right
        # 连接根与右子树最小结点
        if right:
            while(right.left):
                right=right.left
            pRootOfTree.right,right.left=right,pRootOfTree
        while(pRootOfTree.left):
            pRootOfTree=pRootOfTree.left
        return pRootOfTree
