## stack
```
class Solution:
    def Convert(self, pRootOfTree):
        if not pRootOfTree: return None
        pre = temp = TreeNode(0)
        cur = pRootOfTree
        stack = []
        while cur or stack:
            while cur:
                stack.append(cur)
                cur = cur.left
            node = stack.pop()
            node.left = pre
            pre.right = node
            pre = node
            cur = node.right
        ans = temp.right
        ans.left = None
        return ans
```
