class Solution:
    def GetNext(self, pNode):
        if pNode.right:
            node = pNode.right
            while node.left: node = node.left
            return node
        else:
            if not pNode.next: return None
            else:
                if pNode.next.left == pNode:
                    return pNode.next
                else:
                    while pNode.next.left != pNode:
                        pNode = pNode.next
                        if not pNode.next: return None
                    return pNode.next
