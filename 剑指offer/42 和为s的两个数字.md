class Solution:
    def FindNumbersWithSum(self, array, tsum):
        # write code here
        start, end = 0, len(array) - 1
        while start < end:
            val = array[start] + array[end]
            if val == tsum:
                return [array[start], array[end]]
            elif val < tsum:
                start += 1
            else:
                end -= 1
        return []
