class Solution:
    def Serialize(self, root):
        if not root: return '$'
        return str(root.val) + ',' + self.Serialize(root.left) + ',' + self.Serialize(root.right)
    def Deserialize(self, s):
        # write code here
        s = s.split(',')
        return self.f(s)
    def f(self, s):
        val = s.pop(0)
        if val == '$': 
            return None
        else:
            node = TreeNode(int(val))
            node.left = self.f(s)
            node.right = self.f(s)
            return node
