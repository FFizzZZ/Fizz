class Solution:
    def HasSubtree(self, pRoot1, pRoot2):
        # write code here
        if pRoot1 and pRoot2:
            return self.f(pRoot1, pRoot2) or self.HasSubtree(pRoot1.left, pRoot2) or self.HasSubtree(pRoot1.right, pRoot2)
        return False
    def f(self, node, p):
        if not p:
            return True
        if not node:
            return False
        if node.val != p.val:
            return False
        return self.f(node.left, p.left) and self.f(node.right, p.right)
