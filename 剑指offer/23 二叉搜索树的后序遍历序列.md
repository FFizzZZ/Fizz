```
class Solution:
    def VerifySquenceOfBST(self, sequence):
        if not sequence: return False
        def f(num):
            if len(num) <= 2: return True
            value = num[-1]
            flag = 0
            for i in range(len(num)-1):
                if num[i] > value:
                    flag = 1
                    break
            i = i if flag else -1
            if num[i:-1] and min(num[i:-1]) < value:
                return False
            return f(num[:i]) and f(num[i:-1])
        return f(sequence)
```

## recursive
```
class Solution:
    def VerifySquenceOfBST(self, sequence):
        if not sequence: return False
        n = len(sequence)
        root = sequence[-1]
        for i, val in enumerate(sequence):
            if val > root:
                break
        for j in range(i, n):
            if sequence[j] < root:
                return False
        left = right = True
        if i > 0:
            left = self.VerifySquenceOfBST(sequence[:i])
        if i < n - 1:
            right = self.VerifySquenceOfBST(sequence[i:n - 1])
        return left and right
```
