class Solution:
    def minNumberInRotateArray(self, rotateArray):
        if not rotateArray: return 0
        if rotateArray[0] < rotateArray[-1]:
            return rotateArray[0]
        l, r = 0, len(rotateArray) - 1
        while l < r:
            mid = (l + r) / 2
            val = rotateArray[mid]
            if val >= rotateArray[0]:
                l = mid + 1
            else:
                r = mid
        return rotateArray[l]
