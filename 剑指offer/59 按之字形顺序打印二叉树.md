class Solution:
    def Print(self, pRoot):
        if not pRoot: return []
        ans = []
        index = 1
        stack = [[pRoot], []]
        while any(stack):
            val = []
            if index:
                while stack[0]:
                    node = stack[0].pop()
                    val.append(node.val)
                    if node.left: stack[1].append(node.left)
                    if node.right: stack[1].append(node.right)
            else:
                while stack[1]:
                    node = stack[1].pop()
                    val.append(node.val)
                    if node.right: stack[0].append(node.right)
                    if node.left: stack[0].append(node.left)
            ans.append(val)
            index = 1 - index
        return ans
