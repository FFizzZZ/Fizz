class Solution:
    def Print(self, pRoot):
        if not pRoot: return []
        ans = []
        count = 1
        stack = [pRoot]
        while stack:
            temp = []
            val = []
            for i in stack:
                val.append(i.val)
                if i.left: temp.append(i.left)
                if i.right: temp.append(i.right)
            if count & 1 == 1: ans.append(val)
            else: ans.append(val[::-1])
            stack = temp
            count += 1
        return ans
