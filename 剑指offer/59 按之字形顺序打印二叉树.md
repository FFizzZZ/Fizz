class Solution:
    def Print(self, pRoot):
        if not pRoot: return []
        ans = []
        count = 1
        stack = [pRoot]
        while stack:
            temp = []
            val = []
            if count:
                for i in range(len(stack)-1, -1, -1):
                    node = stack[i]
                    val.append(node.val)
                    if node.left: temp.append(node.left)
                    if node.right: temp.append(node.right)
            else:
                for i in range(len(stack)-1, -1, -1):
                    node = stack[i]
                    val.append(node.val)
                    if node.right: temp.append(node.right)
                    if node.left: temp.append(node.left)
            ans.append(val)
            stack = temp
            count = 1 - count
        return ans
