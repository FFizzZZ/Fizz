class Solution:
    def FindGreatestSumOfSubArray(self, array):
        if not array: return 0
        temp = 0
        ans = array[0]
        for i in array:
            if temp < 0:
                temp = i
            else:
                temp += i
            ans = max(temp, ans)
        return ans
