class Solution:
    def FindGreatestSumOfSubArray(self, array):
        if not array: return 0
        dp = []
        self.f(len(array)-1, array, dp)
        return max(dp)
    def f(self, end, array, dp):
        if end == 0: return array[0]
        ans = self.f(end-1, array, dp) + array[end] if self.f(end-1, array, dp) >= 0 else array[end]
        dp.append(ans)
        return ans
