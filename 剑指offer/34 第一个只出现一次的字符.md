class Solution:
    def FirstNotRepeatingChar(self, s):
        # write code here
        if not s: return -1
        d = {}
        for i in s:
            d[i] = d.setdefault(i, 0) + 1
        val = list(d.items())
        index = [x[0] for x in val if x[1] == 1]
        return min([s.index(x) for x in index])
