```
class Solution:
    def multiply(self, A):
        l, u, n, B = [1], [1], len(A), []
        for i in range(n-1):
            l.append(l[-1] * A[i])
            u.append(u[-1] * A[~i])
        for i in range(n):
            B.append(l[i] * u[~i])
        return B
```

## 节省空间
```
class Solution:
    def multiply(self, A):
        B, n = [1], len(A)
        for i in range(1, n):
            B.append(B[-1] * A[i-1])
        temp = 1
        for i in range(n-2, -1, -1):
            temp *= A[i+1]
            B[i] *= temp
        return B
```
