class Solution:
    def IsBalanced_Solution(self, pRoot):
        # write code here
        if not pRoot: return True
        return self.IsBalanced_Solution(pRoot.left) and self.IsBalanced_Solution(pRoot.right) and abs(self.f(pRoot.left)-self.f(pRoot.right)) <= 1
    def f(self, node):
        if not node: return 0
        return max(self.f(node.right) + 1, self.f(node.left) + 1)
