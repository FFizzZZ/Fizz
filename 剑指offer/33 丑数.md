class Solution:
    def GetUglyNumber_Solution(self, index):
        # write code here
        if not index: return 0
        ans = [1]
        next_index = 1
        index2, index3, index5 = 0, 0, 0
        for i in range(index-1):
            val = min(ans[index2] * 2, ans[index3] * 3, ans[index5] * 5)
            ans.append(val)
            while ans[index2] * 2 <= val:
                index2 += 1
            while ans[index3] * 3 <= val:
                index3 += 1
            while ans[index5] * 5 <= val:
                index5 += 1
        return ans[-1]
