class Solution:
    def IsContinuous(self, numbers):
        # write code here
        if not numbers: return False
        numbers.sort()
        l = numbers.count(0)
        num = [x for x in numbers if x != 0]
        count = 0
        for i in range(1, len(num)):
            diff = num[i] - num[i-1]
            if diff == 0: return False
            if diff != 1:
                count += diff - 1
        return True if count <= l else False
