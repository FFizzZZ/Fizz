class Solution:
    def TreeDepth(self, pRoot):
        # write code here
        depth = []
        self.f(pRoot, 0, depth)
        return max(depth)
    def f(self, node, count, depth):
        if not node: 
            depth.append(count)
            return
        self.f(node.right, count+1, depth)
        self.f(node.left, count+1, depth)

class Solution:
    def TreeDepth(self, pRoot):
        return max(1 + self.TreeDepth(pRoot.left), 1 + self.TreeDepth(pRoot.right)) if pRoot else 0
