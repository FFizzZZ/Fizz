class Solution:
    def EntryNodeOfLoop(self, pHead):
        p1, p2 = pHead, pHead
        flag = 0
        while p1.next:
            p1, p2 = p1.next.next, p2.next
            if p1 == p2:
                flag = 1
                break
        if not flag: return None
        else:
            p1, count = p1.next, 1
            while p1 != p2:
                p1, count = p1.next, count + 1
            p1, p2 = pHead, pHead
            for i in range(count): p1 = p1.next
            while p1 != p2:
                p1, p2 = p1.next, p2.next
            return p1
