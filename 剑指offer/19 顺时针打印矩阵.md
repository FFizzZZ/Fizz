class Solution:
    # matrix类型为二维列表，需要返回列表
    def printMatrix(self, matrix):
        # write code here
        ans = []
        r, c = len(matrix), len(matrix[0])
        rd = [0, 1, 0, -1]
        cd = [1, 0, -1, 0]
        d = {}
        i, j = 0, 0
        dr, dc = 0, 0
        while len(d) < r * c:
            if 0 <= i < r and 0 <= j < c and (i, j) not in d:
                d[(i, j)] = 1
                ans.append(matrix[i][j])
            else:
                i -= rd[dr]
                j -= cd[dc]
                dr = (dr + 1) % 4
                dc = (dc + 1) % 4
            i = i + rd[dr]
            j = j + cd[dc]
        return ans
