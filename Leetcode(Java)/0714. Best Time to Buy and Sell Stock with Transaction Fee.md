class Solution {
    public int maxProfit(int[] prices, int fee) {
        int sell = 0, buy = Integer.MIN_VALUE;
        for (int price : prices) {
            int tmp = buy;
            buy = Math.max(sell - price, buy);
            sell = Math.max(sell, tmp + price > fee? tmp + price - fee : 0);
        }
        return sell;
    }
}

class Solution {
    public int maxProfit(int[] prices, int fee) {
        int cash = 0, hold = -prices[0];
        for (int i = 1; i < prices.length; i++) {
            cash = Math.max(cash, hold + prices[i] - fee);
            hold = Math.max(hold, cash - prices[i]);
        }
        return cash;
    }
}

class Solution {
    public int maxProfit(int[] prices, int fee) {
        int ans = 0, low = prices[0];
        for (int price : prices) {
            if (price < low)
                low = price;
            else if (price - low - fee > 0) {
                ans += price - low - fee;
                low = price - fee;
            }
        }
        return ans;
    }
}
