class Solution {
    public void sortColors(int[] nums) {
        int zero = -1, two = nums.length;
        for (int i = 0; i < nums.length; i++) {
            if (i == two) break;
            while (nums[i] == 2 && i < two)
                swap(nums, i, --two);
            if (nums[i] == 0)
                swap(nums, i, ++zero);
        }
    }
    
    private void swap(int[] nums, int i, int j) {
        int tmp = nums[i]; nums[i] = nums[j]; nums[j] = tmp;
    }
}
