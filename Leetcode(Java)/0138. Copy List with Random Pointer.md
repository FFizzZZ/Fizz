## Iteration
```
class Solution {
    public Node copyRandomList(Node head) {
        if (head == null) return head;
        Node node = head, tmp;
        while (node != null) {
            tmp = new Node(node.val, node.next, null);
            node.next = tmp;
            node = tmp.next;
        }
        node = head;
        while (node != null) {
            tmp = node.next;
            if (node.random != null) tmp.random = node.random.next;
            node = tmp.next;
        }
        node = head; tmp = head.next;
        Node ans = head.next;
        while (node != null) {
            node.next = tmp.next;
            node = node.next;
            if (node != null) {
                tmp.next = node.next;
                tmp = tmp.next;
            }
        }
        return ans;
    }
}
```
