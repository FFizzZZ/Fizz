## Iteration
```
class Solution {
    public Node flatten(Node head) {
        if (head == null) return head;
        Node node = head;
        while (node != null) {
            if (node.child == null) {
                node = node.next;
                continue;
            }
            Node tmp = node.child;
            while (tmp.next != null) tmp = tmp.next;
            tmp.next = node.next;
            if (node.next != null) node.next.prev = tmp;
            node.next = node.child;
            node.child.prev = node;
            node.child = null;
        }
        return head;
    }
}
```
