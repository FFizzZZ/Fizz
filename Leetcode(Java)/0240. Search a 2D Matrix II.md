class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        if (matrix == null || matrix.length == 0) return false;
        int i = matrix.length - 1, j = 0, n = matrix[0].length;
        while (i > -1 && j < n) {
            int val = matrix[i][j];
            if (val == target) return true;
            else if (val < target) j += 1;
            else i -= 1;
        }
        return false;
    }
}
