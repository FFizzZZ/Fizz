class Solution {
    public int countPrimes(int n) {
        boolean[] prime = new boolean[n];
        Arrays.fill(prime, true);
        int count = 0;
        for (int i = 2; i * i <= n; i++) {
            if (prime[i] == true) {
                for (int j = i; i * j < n; j++) {
                    prime[i * j] = false;
                }
            }
        }
        for (int i = 2; i < n; i++)
            if (prime[i] == true) count++;
        return count;
    }
}
