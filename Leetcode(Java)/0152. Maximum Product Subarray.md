class Solution {
    public int maxProduct(int[] nums) {
        int ans = nums[0];
        int max = 1, min = 1, tmp;
        for (int x : nums) {
            if (x < 0) {
                tmp = max; max = min; min = tmp;
            }
            max = x > x * max? x : x * max;
            min = x < x * min? x : x * min;
            ans = Math.max(ans, max);
        }
        return ans;
    }
}
