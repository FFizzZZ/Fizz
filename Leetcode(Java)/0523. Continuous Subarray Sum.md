class Solution {
    public boolean checkSubarraySum(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int cur = 0;
        for (int i = 0; i < nums.length; i++) {
            cur += nums[i];
            if (k != 0) cur %= k;
            Integer pre = map.get(cur);
            if (pre != null) {
                if (i - pre > 1) return true;
            }
            else
                map.put(cur, i);
        }
        return false;
    }
}
