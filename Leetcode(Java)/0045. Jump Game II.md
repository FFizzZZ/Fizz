class Solution {
    public int jump(int[] nums) {
        int end = 0, reach = 0, count = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (i + nums[i] > reach) reach = i + nums[i];
            if (reach >= nums.length - 1) return count + 1;
            if (i == end) {
                count += 1;
                end = reach;
            }
        }
        return count;
    }
}
