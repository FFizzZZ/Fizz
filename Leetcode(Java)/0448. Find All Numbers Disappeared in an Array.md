## Negative
```
class Solution {
    public List<Integer> findDisappearedNumbers(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < nums.length; i++) {
            int idx = Math.abs(nums[i]) - 1;
            if (nums[idx] > 0)
                nums[idx] *= -1;
        }
        for (int i = 0; i < nums.length; i++) 
            if (nums[i] > 0)
                ans.add(i + 1);
        return ans;
    }
}
```

## Add
```
class Solution {
    public List<Integer> findDisappearedNumbers(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        int N = nums.length + 1;
        for (int i = 0; i < N - 1; i++)
            nums[(nums[i] % N) - 1] += N;
        for (int i = 0; i < N - 1; i++)
            if (nums[i] < N)
                ans.add(i + 1);
        return ans;
    }
}
```
