## Leetcode

#### Union-Find
```
class DSU:
    def __init__(self, N):
        self.p = list(range(N))
        
    def find(self, x):
        if self.p[x] != x:
            self.p[x] = self.find(self.p[x])
        return self.p[x]
    
    def union(self, x, y):
        xr = self.find(x)
        yr = self.find(y)
        self.p[xr] = yr
        
class Solution:
    def largestComponentSize(self, A):
        B = list()
        for x in A:
            factors = list()
            d = 2
            while d * d <= x:
                if x % d == 0:
                    while x % d == 0:
                        x //= d
                    factors.append(d)
                d += 1
            if x > 1 or not factors:
                factors.append(x)
            B.append(factors)

        primes = list(set(x for b in B for x in b))
        prime_to_index = {p: i for i, p in enumerate(primes)}
        dsu = DSU(len(primes))
        for b in B:
            for x in b:
                dsu.union(prime_to_index[b[0]], prime_to_index[x])
        count = collections.Counter(dsu.find(prime_to_index[factors[0]]) for factors in B)
        return max(count.values())
```
